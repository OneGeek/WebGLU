324             // Find and initialize all uniforms and attributes found in the source
325             this._parseShaderVariables = function(str) {
326                 var tokens = str.split(/[\s\n;]+?/);
327 
328                 for (var i = 0; i < tokens.length; i++) {
329                     if (tokens[i] == "attribute") {
330                         var type = tokens[i+1];
331                         var name = tokens[i+2];
332                         var length = this.types[type];
333                         this.addAttribute(name, length);
334                     }                               
335                     if (tokens[i] == "uniform") {
336                         var type = tokens[i+1];
337                         var name = tokens[i+2];
338                         this.addUniform(name);
339 
340                         if (name == $W.constants.ModelViewUniform) {
341                             this.setModelViewUniform($W.constants.ModelViewUniform);
342 
343                         }else if (name == $W.constants.ProjectionUniform) {
344                             this.setProjectionUniform($W.constants.ProjectionUniform);
345 
346                         }else if (name == $W.constants.NormalMatrixUniform) {
347                             this.setNormalMatrixUniform($W.constants.NormalMatrixUniform);
348                         }
349                     }
350                 }
351             }

353             /** Compile the shader if able.
354              * Lets any shader programs which use this shader know they need to
355              * relink
356              */
357             this.compile = function() {
358                 if (this.type == $W.GL.VERTEX_SHADER) {
359                     console.log("compiling '" + this.name + "' as vertex shader");
360                 }else if (this.type = $W.GL.FRAGMENT_SHADER) {
361                     console.log("compiling '" + this.name + "' as vertex shader");
362                 }else {
363                     console.error("compiling '" + this.name + "' as unknown type " + this.type);
364                 }   
365                 if (this.glShader !== null) {
366                     $W.GL.deleteShader(this.glShader);
367                     this.glShader = null;
368                 }
369 
370                 var shader = $W.GL.createShader(this.type);
371                 
372                 $W.GL.shaderSource(shader, this.source);
373                 $W.GL.compileShader(shader);
374 
375                 if (!$W.GL.getShaderParameter(shader, $W.GL.COMPILE_STATUS)) {
376                     console.group('Compile error');
377                     //console.error('error compiling ' + this.name + ': ' + $W.GL.getShaderInfoLog(shader));
378                     console.log(this.source);
379                     console.groupEnd();
380                     this.glShader = null;
381                 } else {
382                     this._clean();
383                     this.glShader = shader;
384                 }
385 
386                 return (this.glShader !== null);
387             }

603             /** Link this shader program to make it useable.
604              * Will [re]compile all attached shaders if necessary.
605              */
606             this.link = function() {
607                 console.groupCollapsed("linking '" + this.name + "'");
608 
609                 if (this.isLinked) {
610                     console.log("already exists, deleting and relinking");
611 
612                     // Only delete the program if one already exists
613                     if (!!this.glProgram) {
614                         $W.GL.deleteProgram(this.glProgram);
615                         this.attributes = [];
616                         this.glProgram = null;
617                     }
618                 }
619 
620                 this.glProgram = $W.GL.createProgram();
621 
622                 // Attach all the shaders
623                 for (var i = 0; i < this.shaders.length; i++) {
624                     var shader = $W.shaders[this.shaders[i]];
625                     var glShader = shader.getGLShader();
626 
627                     // if the shader is still dirty after calling get,
628                     // which should have cleaned it, then the compile failed.
629                     if (shader.isDirty) {
630                         console.log(this.shaders[i] + " failed to compile");
631                     }else {
632                         $W.GL.attachShader(this.glProgram, 
633                                 $W.shaders[this.shaders[i]].glShader);
634                     }	
635                 }
636 
637                 $W.GL.linkProgram(this.glProgram);
638 
639                 // Check for errors
640                 if (!$W.GL.getProgramParameter(this.glProgram, $W.GL.LINK_STATUS)) {
641                     console.group('Link error');
642                     console.error($W.GL.getProgramInfoLog(this.glProgram));
643                     this.isLinked = false;
644                     console.groupEnd();
645                 }
646                     
647                 this.isLinked = true;
648                 this.isDirty = false;
649                 this._setupAttributes();
650                 this._setupUniforms();
651 
652                 console.groupEnd();
653                 return this.isLinked;
654             }

2668             var j0 = faces[i+0];

2684             if (!nvecs[j2]) nvecs[j2] = [];

2686             nvecs[j0].push(n);

2688             nvecs[j2].push(n);
2689         }
2690     }

2692     var normals = new Array(vertices.length);
2693 
2694     // now go through and average out everything
2695     for (var i = 0; i < nvecs.length - 1; i++) {
2696         var count = nvecs[i].length;
2697         var x = 0;
2698         var y = 0;
2699         var z = 0;
2700

2703             y += nvecs[i][j].elements[1];
2704             z += nvecs[i][j].elements[2];
2705         }
2706 
2707         normals[i] = [x/count, y/count, z/count];
2708     }
2709 
2710     return normals;
